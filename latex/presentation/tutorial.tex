\section{基本}
\subsection{公式}
\begin{frame}
        \begin{block}{旋转不变性} 
            \begin{equation*}
                LBP^{ri}_{P,R} = min \{ ROR(LBP_{P,R},i) \ | \  i = 0,1,\dots,P-1\} 
            \end{equation*}
        \end{block}

        \begin{block}{Uniform Pattens} 
            \begin{equation*}
                LBP^{riu2}_{P,R} = 
                \begin{cases}
                    \sum^{P-1}_{p=0} s(g_p - g_c) & if \  U(LBP_{P,R}) \leq 2 \\
                    P+1 & otherwise
                \end{cases} 
            \end{equation*}
        \end{block}
\end{frame}
